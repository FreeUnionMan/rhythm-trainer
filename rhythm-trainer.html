    <script>

document.addEventListener('DOMContentLoaded', () => {

            const notesContainer = document.getElementById('notesContainer');

            const newRhythmButton = document.getElementById('newRhythmButton');

            const colorPickerButton = document.getElementById('colorPickerButton');

            const colorPickerContainer = document.getElementById('colorPicker');


            const noteCount = 16;

            const notesPerGroup = 4;

            let selectedColor = 'orange';


            function createNoteGroup() {

                const noteGroup = document.createElement('div');

                noteGroup.classList.add('note-group');

                return noteGroup;

            }


            function createNote() {

                const noteContainer = document.createElement('div');

                noteContainer.classList.add('note-container');


                const note = document.createElement('div');

                note.classList.add('note');

                note.addEventListener('click', toggleNoteColor);


                const noteStem = document.createElement('div');

                noteStem.classList.add('note-stem');


                const noteFlag = document.createElement('div');

                noteFlag.classList.add('note-flag');


                const noteFlagBottom = document.createElement('div');

                noteFlagBottom.classList.add('note-flag-bottom');


                noteContainer.appendChild(note);

                noteContainer.appendChild(noteFlag);

                noteContainer.appendChild(noteStem);

                noteContainer.appendChild(noteFlagBottom);


                return noteContainer;

            }


            function createNotes() {

                notesContainer.innerHTML = '';


                let currentGroup = createNoteGroup();

                notesContainer.appendChild(currentGroup);


                for (let i = 0; i < noteCount; i++) {

                    const noteContainer = createNote();

                    currentGroup.appendChild(noteContainer);


                    if ((i + 1) % notesPerGroup === 0 && i < noteCount - 1) {

                        currentGroup = createNoteGroup();

                        notesContainer.appendChild(currentGroup);

                    }


                    if ((i + 1) % 4 === 0) {

                        const noteFlag = noteContainer.querySelector('.note-flag');

                        noteFlag.classList.add('hidden');


                        const noteFlagBottom = noteContainer.querySelector('.note-flag-bottom');

                        noteFlagBottom.classList.add('hidden');

                    }

                }

            }


            function toggleNoteColor(event) {

                const note = event.target;

                note.style.backgroundColor = note.style.backgroundColor === selectedColor ? 'black' : selectedColor;

            }


            newRhythmButton.addEventListener('click', createNotes);


            colorPickerButton.addEventListener('click', () => {

                colorPickerContainer.classList.toggle('show');

            });


            function createColorOptions() {

                const colors = ['red', 'blue', 'green', 'yellow', 'orange', 'purple', 'pink', 'brown', 'gray', 'black'];

                const colorPicker = document.getElementById('colorPicker');

                let rowCounter = 0;


                for (let i = 0; i < colors.length; i++) {

                    const colorOption = document.createElement('div');

                    colorOption.classList.add('color-option');

                    colorOption.style.backgroundColor = colors[i];

                    colorOption.addEventListener('click', () => {

                        selectedColor = colors[i];

                        colorPickerContainer.classList.remove('show');

                        colorPickerButton.style.backgroundColor = selectedColor;

                    });


                    colorPicker.children[rowCounter].appendChild(colorOption);


                    if ((i + 1) % 3 === 0) {

                        rowCounter++;

                    }

                }

            }


            createNotes();

            createColorOptions();

        });
